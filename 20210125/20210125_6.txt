#include <stdio.h>

int main(){
   char c;
   int i,countNumber=0,countAll=0;

   while((c=getchar())!=EOF){
       countAll++;

       if(c<'0'||c>'9'){

       }
       countNumber++;
   }
   	printf("The number of all characters is:%d",countAll);
   	printf("The number of numbers is:%d",countNumber);
    return 0;
}
