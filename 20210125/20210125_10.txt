/*Упражнение 10.Напишете функция reverse(s), която обръща символния низ
s. Използвайте я, за да напишете програма, която преобръща входа си ред
по ред.*/
#include<stdio.h>

void reverse(void)
{
    char c;
    if((c = getchar()) != '\n'){ 
        reverse(); 
    }
    putchar(c);
    return;
}

int main(void)
{
    printf("Enter a line of text below:n");
    reverse();
    putchar('\n');
}
